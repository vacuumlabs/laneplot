Config

/* data consists of an array of lane objects.*/
data:{
  [{

      /* Optional field for further lane identification. */
      id: "voluntary",

      /* Arbitrary HTML code to render on the left side of the lane */

      description: "<some html>",

      /* Optional label icon */
      icon: "/path/to",

      /* List of lane intervals and one time events */
      items: [
        /* Example interval definition */
        {
          /* Beginning of the interval */
          startDate: "date-string",

          /* End of the interval */
          endDate: "date-string",

          /* Optional field to mark the */
          progressDate: "date-string",

          /* CSS classes of interval */
          classes: ["red", "important"],

          /* Optional event icon */
          icon: "/path/to",

          /* Optional. Progress color (simple color = '#dedede',
            gradient color = ['#dedede', '#ffffff'])
            */
          progressColor: ["#dedede", "#ffffff"],

          /* Optional. Unfinished color (simple color = '#dedede',
            gradient color = ['#dedede', '#ffffff'])
          */
          unfinishedColor: "#ffffff",

          /* Optional text to display inside interval */
          label: "Prozac"
        },

        /* Example one time event definition */
        {

          date: "date-string",

          /* CSS classes of interval */
          classes: ["green", "boring"],

          /* Optional event icon */
          icon: "/path/to",

          /* Optional text to display on the right side of the event */
          label: "EEG"
        }
      ]
    },

    settings: {
      /**
       * Describes the leftmost and rightmost date that can be scrolled to.
       * Both startDate and endDate are optional, and default to the
         first and last date used in items object.
        */
      boundaries: {
        /* Optional, describes leftmost date on the time scale */
        startDate: "date-string",
        /* Optional, describes rightmost date on the time scale */
        endDate: "date-string"
      },

      viewport: {
        /* Width in miliseconds */
        width: 1000 * 60 * 60 * 24 * 30,
        /* Optional date to center the viewport. Defaults to the leftmost boundary. */
        position: null
      },

      grid: {
         /* Optional parameter, default value is #dedede. Sets the color of the grid lines. */
         color: "#dedede",
        /*
         *  Sets timelines at X axis of the grid
         *  Default timeline:
         *      resolution = viewport.width / 5
         *      defateFormat = MM-DD
         */
        timelines: [
            {
                /* The interval in miliseconds in which a vertical gridline is drawn. */
                resolution: 1000 * 60 * 60 * 24 /* 1 day */,
                /*Sets date format of timeline */
                dateFormat: "MM-DD"
            }
        ]
       }
      /* Height of the single lane in pixels */
      laneHeight: 50
    },

    /* Optional event handler of taps/clicks on lane */
    onLaneClick: function (lane) { /* Do something */ },
    onError: function (errorObj, targetObj) { /* Do something */ }
  };
